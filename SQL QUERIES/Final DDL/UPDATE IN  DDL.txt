NOTIFICATION AND USER_NOTIFICATION TABLE DELETED!

---------------------------------------------------------------------------------
CREATE TABLE NOTIFICATION
(
	NOTIFICATION_ID INTEGER PRIMARY KEY,
	FROM_ID INTEGER NOT NULL,
	TO_ID INTEGER NOT NULL,
	CONTENT VARCHAR2(20),
	NOTIFIED_TIME DATE default SYSDATE, 
	IS_SEEN INTEGER DEFAULT 0,

	FOREIGN KEY (FROM_ID) REFERENCES USERACCOUNT(USER_ID) ON DELETE CASCADE,
	FOREIGN KEY (TO_ID) REFERENCES  USERACCOUNT(USER_ID) ON DELETE CASCADE
);

ALTER TABLE NOTIFICATION
ADD RELATED_POST_ID INTEGER;

ALTER TABLE NOTIFICATION
ADD CONSTRAINT FK_NOTIFY_POST
FOREIGN KEY (RELATED_POST_ID) REFERENCES POST(POST_ID) ON DELETE CASCADE;

------ADDED TRIGGER FOR  GENERATING NOTIFTICATION PK------------

CREATE SEQUENCE notification_seq START WITH 1;

CREATE OR REPLACE TRIGGER notification_on_insert
BEFORE INSERT ON NOTIFICATION 
FOR EACH ROW

BEGIN
  SELECT notification_seq.NEXTVAL
  INTO   :new.NOTIFICATION_ID
  FROM   dual;
END;
