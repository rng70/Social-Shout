
CREATE TABLE CHAT (
    FROM_ID INTEGER NOT NULL,
    TO_ID INTEGER NOT NULL,
    TEXT VARCHAR2(3000) ,
		SENT_TIME DATE DEFAULT SYSDATE,
		IS_SEEN INTEGER DEFAULT 0,
		
    FOREIGN KEY (FROM_ID) REFERENCES USERACCOUNT(USER_ID) ON DELETE CASCADE,
		FOREIGN KEY (TO_ID) REFERENCES  USERACCOUNT(USER_ID) ON DELETE CASCADE

);


DROP TRIGGER UNFOLLOW_USER_TRIGGER;
